hexo搭建小记

###安装

npm install hexo-cli -g

###初始化

hexo init blog
cd blog
npm install

###新建文章

hexo new "hexo"

###git部署

npm install hexo-deployer-git --save

修改_config.yml #Deployment
deploy:
  type: git
  repo: https://github.com/username/username.github.io.git
  branch: master

hexo d -g
